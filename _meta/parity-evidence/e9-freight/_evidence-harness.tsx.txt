function dump(name: string, html: string) {
  fs.mkdirSync(OUT, { recursive: true });
  fs.writeFileSync(path.join(OUT, `${name}.html`), html, 'utf-8');
}

describe('E9 parity evidence — capture per-state DOM', () => {
  it('carriers: loading / table / empty / denied / error / dialog (optimistic)', async () => {
    // loading
    const slow = render(<CarriersView labels={CARRIER_LABELS} listCarriersAction={() => new Promise(() => {})} upsertCarrierAction={vi.fn()} listLanesAction={vi.fn()} upsertLaneAction={vi.fn()} />);
    dump('carriers-01-loading', slow.container.innerHTML);
    slow.unmount();

    // table
    const ok = render(<CarriersView labels={CARRIER_LABELS} listCarriersAction={vi.fn(async () => [CARRIER])} upsertCarrierAction={vi.fn(async () => ({ ok: true as const, data: CARRIER }))} listLanesAction={vi.fn(async () => [LANE])} upsertLaneAction={vi.fn(async () => ({ ok: true as const, data: LANE }))} />);
    await waitFor(() => screen.getByTestId('carriers-table'));
    dump('carriers-02-table', ok.container.innerHTML);
    // dialog (optimistic submit pending captured as the open form)
    fireEvent.click(screen.getByTestId('carriers-add'));
    await waitFor(() => screen.getByTestId('carrier-form'));
    dump('carriers-03-dialog', document.body.innerHTML);
    // lanes panel
    fireEvent.click(screen.getByTestId('carrier-cancel'));
    fireEvent.click(screen.getByTestId('carrier-lanes-DHL'));
    await waitFor(() => screen.getByTestId('lanes-table'));
    dump('carriers-04-lanes-panel', ok.container.innerHTML);
    ok.unmount();

    // empty
    const empty = render(<CarriersView labels={CARRIER_LABELS} listCarriersAction={vi.fn(async () => [])} upsertCarrierAction={vi.fn()} listLanesAction={vi.fn()} upsertLaneAction={vi.fn()} />);
    await waitFor(() => screen.getByTestId('carriers-empty'));
    dump('carriers-05-empty', empty.container.innerHTML);
    empty.unmount();

    // denied
    const denied = render(<CarriersView labels={CARRIER_LABELS} listCarriersAction={vi.fn(async () => { throw new Error('forbidden'); })} upsertCarrierAction={vi.fn()} listLanesAction={vi.fn()} upsertLaneAction={vi.fn()} />);
    await waitFor(() => screen.getByTestId('carriers-denied'));
    dump('carriers-06-denied', denied.container.innerHTML);
    denied.unmount();

    // error
    const err = render(<CarriersView labels={CARRIER_LABELS} listCarriersAction={vi.fn(async () => { throw new Error('boom'); })} upsertCarrierAction={vi.fn()} listLanesAction={vi.fn()} upsertLaneAction={vi.fn()} />);
    await waitFor(() => screen.getByTestId('carriers-error'));
    dump('carriers-07-error', err.container.innerHTML);
    err.unmount();
  });

  it('scorecard: KPIs + recent table / empty', () => {
    const full = render(<ScorecardView scorecard={SCORECARD} labels={SCORECARD_LABELS} />);
    dump('scorecard-01-kpis-and-recent', full.container.innerHTML);
    full.unmount();

    const empty = render(<ScorecardView scorecard={{ onTimePct: null, avgQtyVariancePct: null, ncrCount: 0, openNcrCount: 0, recentPos: [] }} labels={SCORECARD_LABELS} />);
    dump('scorecard-02-empty', empty.container.innerHTML);
    empty.unmount();
  });
});
